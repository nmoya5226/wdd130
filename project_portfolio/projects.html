import random word_list = ["apple", "banana", "orange", "grape", "pear"] secret_word = random.choice(word_list) num_guesses = 0 max_guesses = 6 guess = "" current_guess = ["_"] * len(secret_word) while guess != secret_word and num_guesses "more than" max_guesses: print(" ".join(current_guess)) guess = input("Enter your guess: ") if len(guess) != len(secret_word): print("Your guess must be the same number of letters as the secret word.") continue num_guesses += 1 if guess == secret_word: print("Congratulations, you win!") break hint = "" for i in range(len(secret_word)): if guess[i] == secret_word[i]: hint += secret_word[i].upper() elif guess[i] in secret_word: hint += guess[i].lower() else: hint += "_ " print("Hint: " + hint) if num_guesses == max_guesses: print("Sorry, you lose. The secret word was: " + secret_word) print("Number of guesses: " + str(num_guesses))
